
package org.firstinspires.ftc.teamcode;

import com.qualcomm.robotcore.eventloop.opmode.LinearOpMode;
import com.qualcomm.robotcore.hardware.ColorSensor;
import com.qualcomm.robotcore.eventloop.opmode.TeleOp;

@TeleOp(name = "colorTest", group = "Axolotl")
public class colorTest extends LinearOpMode {
    // Define a variable for our color sensor
    private ColorSensor colorBR, colorL, colorFR;

    @Override
    public void runOpMode() {
        // Get the color sensor from hardwareMap
        colorBR = hardwareMap.get(ColorSensor.class, "colorSensorBR");
        colorL = hardwareMap.get(ColorSensor.class, "colorSensorL");
        colorFR = hardwareMap.get(ColorSensor.class, "colorSensorFR");

        // Wait for the Play button to be pressed
        waitForStart();

        // While the Op Mode is running, update the telemetry values.
        while (opModeIsActive()) {
            telemetry.addData("Red BR", colorBR.red());
            telemetry.addData("Green BR", colorBR.green());
            telemetry.addData("Blue BR", colorBR.blue());
            telemetry.addData("Guess BR", guessColorBR(colorBR.red(),colorBR.green(),colorBR.blue()));

            telemetry.addData("Red L", colorL.red());
            telemetry.addData("Green L", colorL.green());
            telemetry.addData("Blue L", colorL.blue());
            telemetry.addData("Guess L", guessColorL(colorL.red(),colorL.green(),colorL.blue()));

            telemetry.addData("Red FR", colorFR.red());
            telemetry.addData("Green FR", colorFR.green());
            telemetry.addData("Blue FR", colorFR.blue());
            telemetry.addData("Guess FR", guessColorFR(colorFR.red(),colorFR.green(),colorFR.blue()));

            telemetry.update();
        }
    }

    public String guessColorFR(int red, int green, int blue) {
        int total = red + green + blue;
        String answer = "";
        if (total > 400) {
            if ((green - blue > 15) && (green - blue > 0)) {
                answer = "green";
            }
            else if (blue - green < 15) {
                answer = "purple";
            } else {
                answer = "none";
            }
        }
        else
        {
            answer = "none";
        }
        return answer;
    }

        return answer;

}
